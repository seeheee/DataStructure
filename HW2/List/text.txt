bool  BinarySearch(int *arr, int  start, int  end, int  key) {
    if (start > end)  return  false; 
    int  mid = (start + end) / 2;
    if (arr[mid] == key) {    
        return  true; 
     else  if (arr[mid] > key) { 
        return  BinarySearch(arr, start, mid - 1, key);   
    } else {  
        return  BinarySearch(arr, mid + 1, end, key);
}
